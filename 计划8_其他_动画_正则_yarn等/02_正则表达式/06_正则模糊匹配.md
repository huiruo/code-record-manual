### filter 和 forEach() 
```js
const rgx = new RegExp(searchLatestWord.current, 'i')

// filter
const res = _.filter(dataSource.current, v => {
	console.log('for_v:',v.index_name)
	console.log('test:',rgx.test(v.index_name))
	console.log('分割线--->')
	return rgx.test(v.index_name)
})

// forEach
const targetArr:any = []
const reg= new RegExp(searchLatestWord.current,'i');
dataSource.current.forEach((v) => {
	const flag = reg.test(v.index_name)
	if (flag) {
		targetArr.push(v)
	}
})
```
